## 技术实现

### 1. DBSCAN 聚类（集结区域识别）
```javascript
function clusterStayAreas(trackPoints, eps = 0.8, minPoints = 3) {
  const stayPoints = trackPoints.filter(p => p.speed < 5 && p.duration > 1800)
  const clusters = []
  const visited = new Set()

  for (const point of stayPoints) {
    if (visited.has(point.id)) continue
    visited.add(point.id)
    const neighbors = getNeighbors(point, stayPoints, eps)
    if (neighbors.length >= minPoints) {
      const cluster = expandCluster(point, neighbors, stayPoints, eps, minPoints, visited)
      clusters.push(cluster)
    }
  }

  return clusters.map(cluster => ({
    center: calculateCenter(cluster),
    radius: calculateRadius(cluster),
    frequency: cluster.length,
    avgDuration: avg(cluster.map(p => p.duration)),
    type: inferAreaType(cluster),
    terrain: majorityTerrain(cluster)
  }))
}

function inferAreaType(cluster) {
  const avgDuration = avg(cluster.map(p => p.duration))
  const terrain = majorityTerrain(cluster)
  if (avgDuration > 21600 && terrain === 'forest_edge') return 'assembly'
  if (avgDuration < 14400 && nearRoad(cluster)) return 'supply'
  if (terrain === 'open_field' && hasFireActivity(cluster)) return 'firing_position'
  return 'general'
}
```

### 2. 异常检测（Z-score 方法）
```javascript
function detectAnomalies(trackHistory, target) {
  const anomalies = []
  const stats = calculateHistoricalStats(trackHistory)

  // 时间异常
  const hourOfActivity = new Date(target.timestamp).getHours()
  const timeZScore = (hourOfActivity - stats.avgActivityHour) / stats.stdActivityHour
  if (Math.abs(timeZScore) > 2.0) {
    anomalies.push({
      type: 'time_anomaly',
      severity: Math.abs(timeZScore) > 3 ? 'high' : 'medium',
      zScore: timeZScore,
      description: `活动时间偏离常规 (Z=${timeZScore.toFixed(1)})`
    })
  }

  // 路线偏离
  const routeDeviation = calculateRouteDeviation(target.position, stats.commonRoutes)
  if (routeDeviation > stats.avgDeviation + 2 * stats.stdDeviation) {
    anomalies.push({
      type: 'route_deviation',
      severity: routeDeviation > stats.avgDeviation + 3 * stats.stdDeviation ? 'high' : 'medium',
      deviation: routeDeviation,
      description: `偏离常用路线 ${routeDeviation.toFixed(1)}km`
    })
  }

  return anomalies
}
```

### 3. 轨迹相似度（DTW + Fréchet距离）
```javascript
function calculateTrackSimilarity(trackA, trackB) {
  const routeOverlap = calculateFrechetDistance(trackA.routes, trackB.routes)
  const timeSimilarity = compareTimePatterns(trackA.timePattern, trackB.timePattern)
  const areaOverlap = compareAreas(trackA.areas, trackB.areas)

  return {
    overall: routeOverlap * 0.4 + timeSimilarity * 0.3 + areaOverlap * 0.3,
    routeOverlap,
    timeSimilarity,
    areaOverlap
  }
}
```
